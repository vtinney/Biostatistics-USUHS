#Page 64 Hypothesis Testing II - Power Curves - R

install.packages("pwr")
library(pwr)
delta <- 20
sigma <- 40
d <- delta/sigma
pwr.t.test(d=d, sig.level=.05, power = .80, type = 'two.sample')

plot(pwr.t.test(d=.5, sig.level=.05, power = .80, type = 'two.sample'))
plot(pwr.t.test(d=.125, sig.level=.05, power = .80, type = 'two.sample'))
plot(pwr.t.test(d=.25, sig.level=.05, power = .80, type = 'two.sample'))
plot(pwr.t.test(d=.75, sig.level=.05, power = .80, type = 'two.sample'))
plot(pwr.t.test(d=1.25, sig.level=.05, power = .80, type = 'two.sample'))
plot(pwr.t.test(d=1, sig.level=.05, power = .80, type = 'two.sample'))
